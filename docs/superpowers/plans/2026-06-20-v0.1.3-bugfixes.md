- [ ] **Step 3: Change `getFavorites` return shape**

In `src/main/services/booruService.ts`, change `getFavorites` to return `PaginatedResult<BooruPost>` and count the same filters used for items:

```ts
export async function getFavorites(
  siteId: number,
  page: number = 1,
  limit: number = 20,
  groupId?: number | null,
  rating?: 'safe' | 'questionable' | 'explicit' | 'all',
): Promise<PaginatedResult<BooruPost>> {
  const db = await getDatabase();
  const offset = Math.max(0, (page - 1) * limit);
  const where: string[] = ['f.siteId = ?'];
  const params: any[] = [siteId];

  if (groupId === null) {
    where.push('f.groupId IS NULL');
  } else if (typeof groupId === 'number') {
    where.push('f.groupId = ?');
    params.push(groupId);
  }
  if (rating && rating !== 'all') {
    where.push('p.rating = ?');
    params.push(rating);
  }

  const whereSql = where.join(' AND ');
  const totalRow = await get<{ total: number }>(db, `
    SELECT COUNT(*) as total
    FROM booru_favorites f
    INNER JOIN booru_posts p ON p.id = f.postId
    WHERE ${whereSql}
  `, params);

  const rows = await all<any>(db, `
    SELECT p.*, f.createdAt as favoritedAt, f.groupId
    FROM booru_posts p
    INNER JOIN booru_favorites f ON p.id = f.postId
    WHERE ${whereSql}
    ORDER BY f.createdAt DESC
    LIMIT ? OFFSET ?
  `, [...params, limit, offset]);

  return {
    items: rows.map(mapBooruPostRow),
    total: totalRow?.total ?? 0,
    page,
    limit,
  };
}
```

Add this helper immediately above `getFavorites` and use it in the `items` mapping:

```ts
function mapBooruPostRow(row: any): BooruPost {
  return {
    ...row,
    downloaded: Boolean(row.downloaded),
    isFavorited: Boolean(row.isFavorited),
    isLiked: Boolean(row.isLiked),
    favoriteGroupId: row.favoriteGroupId ?? row.groupId ?? null,
  };
}
```

When known total pages exists, page numbers must include the final page:

```ts
function getKnownPageNumbers(current: number, last: number): Array<number | 'ellipsis'> {
  const pages = new Set<number>([1, last, current, current - 1, current + 1]);
  return Array.from(pages)
    .filter((page) => page >= 1 && page <= last)
    .sort((a, b) => a - b)
    .reduce<Array<number | 'ellipsis'>>((acc, page) => {
      const previous = acc[acc.length - 1];
      if (typeof previous === 'number' && page - previous > 1) {
        acc.push('ellipsis');
      }
      acc.push(page);
      return acc;
    }, []);
}
```
